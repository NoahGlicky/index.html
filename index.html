Alcohol Calculator for a Night Out

Answer how drunk you'd like to get, and we'll calculate how much you'll need to drink based on who is hanging out.

function calculateBottles() { // Get the selected friends and drunk level const friendsSelect = document.getElementById('friends'); const drunkLevelSelect = document.getElementById('drunkLevel'); const loadingDiv = document.getElementById('loading'); const resultDiv = document.getElementById('result'); // Get selected friends (multiple) const selectedFriends = Array.from(friendsSelect.selectedOptions).map(option => option.value); const drunkLevel = parseInt(drunkLevelSelect.value); // Show loading animation loadingDiv.style.display = 'block'; resultDiv.textContent = ''; // Clear any previous results // Fake calculation based on selected friends and drunk level setTimeout(() => { const numberOfFriends = selectedFriends.length; const bottlesNeeded = numberOfFriends * drunkLevel * 0.5; // A basic formula for bottles (this can be adjusted) // Hide loading animation loadingDiv.style.display = 'none'; // Show the result resultDiv.textContent = `You will need approximately ${bottlesNeeded.toFixed(2)} bottles for ${selectedFriends.join(', ')} at drunk level ${drunkLevel}.`; }, 2000); // Simulate a 2-second delay for the calculation }
